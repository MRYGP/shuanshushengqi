
### **推荐算法集成**

```python
# 混合推荐系统
class HybridRecommender:
    def __init__(self):
        self.surprise_model = None  # 协同过滤
        self.content_rules = ContentRules()  # 内容规则
        self.user_profiles = UserProfileDB()
    
    def get_recommendations(self, user_profile):
        # 1. 规则推荐 (60%权重) - 基于用户画像
        rule_recs = self.content_rules.recommend(user_profile)
        
        # 2. 算法推荐 (40%权重) - 基于用户行为
        if self.surprise_model and user_profile.get('user_id'):
            algo_recs = self.get_collaborative_recs(user_profile['user_id'])
        else:
            algo_recs = []
        
        # 3. 混合和排序
        final_recs = self.merge_recommendations(rule_recs, algo_recs)
        return final_recs[:5]
```
